"""Kyber‑768 key exchange (wrapper around reference implementation)."""

try:
    from kyber import Kyber768
except ImportError:
    # Fallback: pure Python implementation from https://github.com/jack4818/kyber-py
    # We include a minimal version inline (not shown for brevity – assume installed)
    raise ImportError("Install kyber-py: pip install kyber-py")

class KyberKeyExchange:
    @staticmethod
    def generate_keypair():
        pk, sk = Kyber768.keygen()
        return pk, sk

    @staticmethod
    def encapsulate(pk):
        return Kyber768.enc(pk)

    @staticmethod
    def decapsulate(sk, ct):
        return Kyber768.dec(ct, sk)

## 2. `flow_encryption/stealth_nn.py` – Lightweight CNN for Stealth

```python
"""Small convolutional network to make ciphertext frames look natural."""

import torch
import torch.nn as nn
import torch.nn.functional as F
import numpy as np

class StealthNet(nn.Module):
    """Tiny CNN (5 layers) that transforms a perturbed frame into a natural‑looking one."""
    def __init__(self):
        super().__init__()
        self.conv1 = nn.Conv2d(3, 16, 3, padding=1)
        self.conv2 = nn.Conv2d(16, 32, 3, padding=1)
        self.conv3 = nn.Conv2d(32, 64, 3, padding=1)
        self.conv4 = nn.Conv2d(64, 32, 3, padding=1)
        self.conv5 = nn.Conv2d(32, 3, 3, padding=1)
        self.bn1 = nn.BatchNorm2d(16)
        self.bn2 = nn.BatchNorm2d(32)
        self.bn3 = nn.BatchNorm2d(64)

    def forward(self, x):
        x = F.relu(self.bn1(self.conv1(x)))
        x = F.relu(self.bn2(self.conv2(x)))
        x = F.relu(self.bn3(self.conv3(x)))
        x = F.relu(self.conv4(x))
        x = torch.sigmoid(self.conv5(x))   # output in [0,1]
        return x

# Pre‑trained model can be loaded from a file; for demo we use identity
def stealth_transform(frame, model=None):
    """Apply neural stealth to a frame (numpy HxWx3 float32)."""
    if model is None:
        # Identity (no change) – in production, load trained weights
        return frame
    with torch.no_grad():
        tensor = torch.from_numpy(frame).permute(2,0,1).unsqueeze(0).float()
        out = model(tensor)
        return out.squeeze(0).permute(1,2,0).cpu().numpy()
```

## 3. `flow_encryption/video.py` – Real‑Time Streaming Cipher

```python
"""Video encryption with adaptive delta, Kyber key exchange, and CNN stealth."""

import cv2
import numpy as np
from .core import KeySchedule, BilinearFlow, adaptive_delta
from .kyber import KyberKeyExchange
from .stealth_nn import StealthNet, stealth_transform
from .aead import AeadStream
import time

class VideoStreamCipher:
    def __init__(self, key, frame_size, delta_base=0.05, use_kyber=False, use_cnn=True):
        """
        key: 256‑bit symmetric key (bytes) – if use_kyber, this is the shared secret.
        frame_size: (width, height) in pixels.
        """
        self.key = key
        self.width, self.height = frame_size
        self.delta_base = delta_base
        self.use_cnn = use_cnn
        if use_cnn:
            self.cnn = StealthNet()
            # In production, load trained weights: self.cnn.load_state_dict(torch.load("stealth.pth"))
            self.cnn.eval()
        else:
            self.cnn = None

        # Generate flow field from key (same as image mode)
        ks = KeySchedule(int.from_bytes(key[:4], 'big'))
        ctrl = ks.random_control_grid(8, 8)
        self.flow = BilinearFlow(ctrl)
        self._ks = ks

        # Precompute per‑pixel random orthonormal basis (for entire frame)
        self._basis = self._precompute_basis()

    def _precompute_basis(self):
        """Compute orthonormal matrix for each pixel (H,W,3,3)."""
        H, W = self.height, self.width
        basis = np.zeros((H, W, 3, 3), dtype=np.float32)
        for i in range(H):
            for j in range(W):
                seed = int.from_bytes(self.key, 'big') + i * W + j
                basis[i,j] = self._ks.random_orthonormal_basis(seed)
        return basis

    def _adaptive_delta(self, frame):
        """Compute per‑pixel delta based on local variance and temporal change."""
        # Simple: use frame gradient magnitude
        gray = cv2.cvtColor(frame, cv2.COLOR_RGB2GRAY)
        grad_x = cv2.Sobel(gray, cv2.CV_32F, 1, 0, ksize=3)
        grad_y = cv2.Sobel(gray, cv2.CV_32F, 0, 1, ksize=3)
        mag = np.sqrt(grad_x**2 + grad_y**2)
        # Normalize and scale
        mag = mag / (mag.max() + 1e-6)
        delta = self.delta_base * (0.5 + mag)
        return np.clip(delta, self.delta_base*0.2, self.delta_base*2.0)

    def encrypt_frame(self, frame):
        """
        frame: numpy array (H,W,3) float32 in [0,1].
        Returns encrypted frame (same shape).
        """
        H, W = self.height, self.width
        # Base flow (evaluate at normalized coordinates)
        y_coords = np.linspace(0, 1, H)
        x_coords = np.linspace(0, 1, W)
        X, Y = np.meshgrid(x_coords, y_coords)
        base = self.flow(X.ravel(), Y.ravel()).reshape(H, W, 3)

        delta_map = self._adaptive_delta(frame)
        delta_3d = delta_map[:,:,np.newaxis]

        # Encrypt: C = B + delta * R · (P - B)
        diff = frame - base
        encrypted = np.zeros_like(frame)
        # Vectorized per pixel (avoid loops – but orthonormal matrix multiply is per pixel)
        # We'll use einsum for speed: C = B + delta * (R @ diff)
        # R is (H,W,3,3), diff is (H,W,3) -> (H,W,3)
        # This is slow in pure Python; we implement a loop for clarity (optimize with numba later)
        for i in range(H):
            for j in range(W):
                R = self._basis[i,j]
                d = diff[i,j]
                enc = base[i,j] + delta_3d[i,j] * (R @ d)
                encrypted[i,j] = np.clip(enc, 0, 1)

        # Apply neural stealth (if enabled)
        if self.use_cnn:
            encrypted = stealth_transform(encrypted, self.cnn)
        return encrypted

    def decrypt_frame(self, encrypted_frame):
        """Inverse transformation."""
        H, W = self.height, self.width
        y_coords = np.linspace(0, 1, H)
        x_coords = np.linspace(0, 1, W)
        X, Y = np.meshgrid(x_coords, y_coords)
        base = self.flow(X.ravel(), Y.ravel()).reshape(H, W, 3)

        delta_map = self._adaptive_delta(encrypted_frame)
        delta_3d = delta_map[:,:,np.newaxis]

        decrypted = np.zeros_like(encrypted_frame)
        for i in range(H):
            for j in range(W):
                R = self._basis[i,j]
                diff = encrypted_frame[i,j] - base[i,j]
                # P = B + (1/delta) * R^T · (C - B)
                dec = base[i,j] + (1.0 / delta_3d[i,j]) * (R.T @ diff)
                decrypted[i,j] = np.clip(dec, 0, 1)
        return decrypted
```

## 4. `flow_encryption/aead.py` – Authenticated Encryption for Key Stream

```python
"""AES‑GCM wrapper to protect the key stream (optional)."""

from cryptography.hazmat.primitives.ciphers.aead import AESGCM
import os

class AeadStream:
    def __init__(self, key):
        self.key = key[:32]   # 256‑bit key

    def encrypt(self, plaintext, associated_data=b''):
        nonce = os.urandom(12)
        cipher = AESGCM(self.key)
        ct = cipher.encrypt(nonce, plaintext, associated_data)
        return nonce + ct

    def decrypt(self, ciphertext, associated_data=b''):
        nonce = ciphertext[:12]
        ct = ciphertext[12:]
        cipher = AESGCM(self.key)
        return cipher.decrypt(nonce, ct, associated_data)
```

## 5. Real‑Time Video Demo (`demo_video.py`)

```python
#!/usr/bin/env python3
"""Live webcam encryption using Flow Encryption v5.0 with Kyber key exchange."""

import cv2
import numpy as np
from flow_encryption.video import VideoStreamCipher
from flow_encryption.kyber import KyberKeyExchange
import pickle
import socket
import threading
import time

def server_side():
    # Receiver (Bob) generates Kyber keypair
    pk, sk = KyberKeyExchange.generate_keypair()
    print("Bob: public key ready. Send to Alice via insecure channel.")
    # In a real app, pk would be sent over network
    # Simulate: Alice receives pk, encapsulates
    ct, shared_secret = KyberKeyExchange.encapsulate(pk)
    print(f"Alice: ciphertext {ct[:16]}..., shared secret derived.")
    # Bob decapsulates
    shared_secret_bob = KyberKeyExchange.decapsulate(sk, ct)
    assert shared_secret == shared_secret_bob
    print("Key exchange successful.")
    return shared_secret

def main():
    # Perform Kyber key exchange (simulated locally)
    shared_secret = server_side()
    # Use first 32 bytes as key
    key = shared_secret[:32]

    # Open webcam
    cap = cv2.VideoCapture(0)
    width = int(cap.get(cv2.CAP_PROP_FRAME_WIDTH))
    height = int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT))
    cipher = VideoStreamCipher(key, (width, height), delta_base=0.05, use_cnn=False)

    print("Press 'e' to encrypt frame, 'd' to decrypt last encrypted, 'q' to quit.")
    encrypted_frame = None
    while True:
        ret, frame = cap.read()
        if not ret:
            break
        frame_rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB) / 255.0
        cv2.imshow("Original", frame)

        key_input = cv2.waitKey(1) & 0xFF
        if key_input == ord('e'):
            encrypted = cipher.encrypt_frame(frame_rgb)
            encrypted_frame = encrypted
            enc_disp = (encrypted * 255).astype(np.uint8)
            cv2.imshow("Encrypted", cv2.cvtColor(enc_disp, cv2.COLOR_RGB2BGR))
        elif key_input == ord('d') and encrypted_frame is not None:
            decrypted = cipher.decrypt_frame(encrypted_frame)
            dec_disp = (decrypted * 255).astype(np.uint8)
            cv2.imshow("Decrypted", cv2.cvtColor(dec_disp, cv2.COLOR_RGB2BGR))
        elif key_input == ord('q'):
            break

    cap.release()
    cv2.destroyAllWindows()

if __name__ == "__main__":
    main()
```
