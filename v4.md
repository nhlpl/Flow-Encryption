## 1. `flow_encryption/core.py` – Orthonormal Basis Generation

```python
"""Core: key schedule, flow fields, and orthonormal basis per pixel."""

import numpy as np
from scipy.ndimage import map_coordinates
from hashlib import pbkdf2_hmac
import hmac
import hashlib

class KeySchedule:
    def __init__(self, key, salt=b'', iterations=100000):
        if isinstance(key, int):
            key = key.to_bytes(32, 'big')
        self.master_key = pbkdf2_hmac('sha256', key, salt, iterations, dklen=32)
        self._counter = 0

    def derive(self, info):
        return hmac.new(self.master_key, info.encode(), hashlib.sha256).digest()

    def random_orthonormal_basis(self, seed):
        """Generate a random 3x3 orthonormal matrix (rotation) from a seed."""
        rng = np.random.RandomState(seed)
        # Random quaternion -> rotation matrix
        q = rng.randn(4)
        q = q / np.linalg.norm(q)
        w, x, y, z = q
        R = np.array([
            [1-2*y*y-2*z*z, 2*x*y-2*w*z, 2*x*z+2*w*y],
            [2*x*y+2*w*z, 1-2*x*x-2*z*z, 2*y*z-2*w*x],
            [2*x*z-2*w*y, 2*y*z+2*w*x, 1-2*x*x-2*y*y]
        ])
        return R.astype(np.float32)

    def random_control_grid(self, h, w):
        rng = np.random.RandomState(int.from_bytes(self.derive("grid")[:4], 'big'))
        return rng.rand(h, w, 3).astype(np.float32)

class BilinearFlow:
    def __init__(self, ctrl_grid):
        self.grid = ctrl_grid
        self.H, self.W = self.grid.shape[:2]

    def __call__(self, x, y):
        x_idx = x * (self.W - 1)
        y_idx = y * (self.H - 1)
        coords = np.stack([y_idx, x_idx], axis=0)
        return map_coordinates(self.grid, coords, order=1, prefilter=False, mode='nearest')
```

```python
class FullColorCipher:
    def __init__(self, key, grid_size=(8,8), delta=0.05, use_fec=True, parallel=True):
        self.key = key
        self.delta = delta
        self.use_fec = use_fec
        self.parallel = parallel
        ks = KeySchedule(key)
        ctrl = ks.random_control_grid(grid_size[0], grid_size[1])
        self.flow = BilinearFlow(ctrl)
        self._ks = ks

    def _encrypt_tile(self, tile, base_tile, delta_tile, seed_offset):
        """Encrypt a single tile (Ht x Wt x 3)."""
        Ht, Wt = tile.shape[:2]
        encrypted = np.zeros_like(tile)
        for i in range(Ht):
            for j in range(Wt):
                seed = seed_offset + i * Wt + j
                R = self._ks.random_orthonormal_basis(seed)
                # Perturbation: C = B + delta * R * (P - B)  (linear, invertible)
                # This is a key‑dependent rotation and scaling around the base color.
                # Actually we want a bijection that hides the plaintext.
                # Use: C = B + delta * R · (P - B)
                diff = tile[i,j] - base_tile[i,j]
                encrypted[i,j] = base_tile[i,j] + delta_tile[i,j] * (R @ diff)
        return np.clip(encrypted, 0, 1)

    def encrypt(self, img_array):
        H, W = img_array.shape[:2]
        # Precompute flow and delta maps
        y_coords = np.linspace(0, 1, H)
        x_coords = np.linspace(0, 1, W)
        X, Y = np.meshgrid(x_coords, y_coords)
        base = self.flow(X.ravel(), Y.ravel()).reshape(H, W, 3)
        delta_map = self.delta * np.ones((H, W))   # could be adaptive
        # Split into tiles for parallel processing
        tile_size = 64
        tiles_h = H // tile_size + (1 if H % tile_size else 0)
        tiles_w = W // tile_size + (1 if W % tile_size else 0)

        def process_tile(tile_idx):
            ti, tj = tile_idx
            y0, y1 = ti*tile_size, min((ti+1)*tile_size, H)
            x0, x1 = tj*tile_size, min((tj+1)*tile_size, W)
            tile = img_array[y0:y1, x0:x1]
            base_tile = base[y0:y1, x0:x1]
            delta_tile = delta_map[y0:y1, x0:x1]
            seed_offset = (ti * tiles_w + tj) * tile_size * tile_size
            return self._encrypt_tile(tile, base_tile, delta_tile, seed_offset)

        if self.parallel:
            with Pool() as pool:
                encrypted_tiles = pool.map(process_tile, [(ti, tj) for ti in range(tiles_h) for tj in range(tiles_w)])
        else:
            encrypted_tiles = [process_tile((ti, tj)) for ti in range(tiles_h) for tj in range(tiles_w)]

        # Reassemble
        encrypted = np.zeros_like(img_array)
        idx = 0
        for ti in range(tiles_h):
            for tj in range(tiles_w):
                y0, y1 = ti*tile_size, min((ti+1)*tile_size, H)
                x0, x1 = tj*tile_size, min((tj+1)*tile_size, W)
                encrypted[y0:y1, x0:x1] = encrypted_tiles[idx]
                idx += 1

        # Optionally apply FEC (embed error correction bits into image)
        if self.use_fec:
            rs = ReedSolomonInterleaved()
            encrypted = rs.encode_image(encrypted)

        # Statistical stealth: make ciphertext histogram match natural image
        encrypted = match_histogram(encrypted)

        tag = compute_hmac(self.key, encrypted.astype(np.float32).tobytes())
        return encrypted, tag
```

## 4. `flow_encryption/stealth.py` – Histogram Matching

```python
"""Make ciphertext histograms match a natural image distribution."""

import numpy as np
from scipy.stats import rankdata

def match_histogram(source, target_hist=None):
    """
    Adjust source image so that its cumulative histogram matches a target.
    If target_hist is None, use a precomputed natural gradient histogram.
    """
    # Flatten each channel
    matched = np.zeros_like(source)
    for c in range(3):
        src = source[:,:,c].flatten()
        if target_hist is None:
            # Use a smooth gradient distribution (e.g., uniform)
            target = np.linspace(0, 1, len(src))
        else:
            target = target_hist[c]
        # Rank-based histogram matching
        sorted_src = np.sort(src)
        indices = rankdata(src, method='ordinal') - 1
        matched[:,:,c] = sorted_src[indices].reshape(source.shape[:2])
    return matched
```

## 5. Web UI (`web.py`) – Drag & Drop

```python
from flask import Flask, render_template, request, send_file
import io
import numpy as np
from PIL import Image
from flow_encryption.image import FullColorCipher

app = Flask(__name__)

@app.route('/', methods=['GET', 'POST'])
def index():
    if request.method == 'POST':
        key = int(request.form['key'])
        delta = float(request.form.get('delta', 0.05))
        file = request.files['image']
        img = Image.open(file).convert('RGB')
        img_array = np.array(img) / 255.0
        cipher = FullColorCipher(key=key, delta=delta)
        enc_array, _ = cipher.encrypt(img_array)
        enc_img = Image.fromarray((enc_array * 255).astype(np.uint8))
        buf = io.BytesIO()
        enc_img.save(buf, format='PNG')
        buf.seek(0)
        return send_file(buf, mimetype='image/png', as_attachment=True, download_name='encrypted.png')
    return '''
```

## 6. Command Line Interface (`cli.py`)

```python
#!/usr/bin/env python3
import argparse
import numpy as np
from PIL import Image
from flow_encryption.image import FullColorCipher

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("mode", choices=["encrypt", "decrypt"])
    parser.add_argument("--input", required=True)
    parser.add_argument("--output", required=True)
    parser.add_argument("--key", type=int, required=True)
    parser.add_argument("--delta", type=float, default=0.05)
    args = parser.parse_args()

    cipher = FullColorCipher(key=args.key, delta=args.delta)
    img = np.array(Image.open(args.input).convert('RGB')) / 255.0
    if args.mode == "encrypt":
        enc, _ = cipher.encrypt(img)
        Image.fromarray((enc * 255).astype(np.uint8)).save(args.output)
    else:
        dec = cipher.decrypt(img)
        Image.fromarray((dec * 255).astype(np.uint8)).save(args.output)

if __name__ == "__main__":
    main()
```
